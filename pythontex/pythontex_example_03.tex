\documentclass[11pt]{article}% 
\usepackage{pythontex} 
\usepackage{nopageno} 
\begin{document} 
\begin{pycode} 
from sympy import * 
x=symbols('x') 
result = latex(integrate("(1+x)**(1/2)",x)) 
\end{pycode} 
 
The result of integrating $\int \sqrt{ 1+x } \, dx$ is given by $\py{result}$ 
 
\end{document}
